//@version=6
indicator("PRO Intraday scalping Indicator ", overlay=true, max_labels_count=500)

// =======================================
// SESSION & DAY LEVELS (INTRADAY)
// =======================================
isNewDay = ta.change(time("D")) != 0

var float dayHigh = na
var float dayLow  = na
var float prevHigh = na
var float prevLow = na

if isNewDay
    prevHigh := dayHigh
    prevLow  := dayLow
    dayHigh := high
    dayLow  := low
else
    dayHigh := math.max(dayHigh, high)
    dayLow  := math.min(dayLow, low)

// =======================================
// PIVOT POINTS (S/R)
// =======================================
pivotHigh = ta.pivothigh(high, 10, 10)
pivotLow  = ta.pivotlow(low, 10, 10)

var float resistLevel = na
var float supportLevel = na

if not na(pivotHigh)
    resistLevel := pivotHigh

if not na(pivotLow)
    supportLevel := pivotLow

plot(resistLevel, title="Resistance", color=color.red, linewidth=1, style=plot.style_stepline)
plot(supportLevel, title="Support", color=color.green, linewidth=1, style=plot.style_stepline)

// =======================================
// STRATEGY: SUPERTREND + MACD (Heavy Duty)
// =======================================
[superTrendLine, superTrendDirection] = ta.supertrend(3, 10)
trendIsUp   = (superTrendDirection == -1)
trendIsDown = (superTrendDirection == 1)

[macdLine, signalLine, histLine] = ta.macd(close, 12, 26, 9)
macdBullish = (macdLine > signalLine)
macdBearish = (macdLine < signalLine)

rsi = ta.rsi(close, 14)

bodySize = math.abs(close - open)
isSignificantCandle = bodySize > ta.tr(true) * 0.2 

// =======================================
// ENTRY CONDITIONS
// =======================================
buySignal = (trendIsUp and macdBullish and histLine > 0 and rsi > 55 and isSignificantCandle)
sellSignal = (trendIsDown and macdBearish and histLine < 0 and rsi < 45 and isSignificantCandle)

// =======================================
// PLOTS
// =======================================
plot(superTrendLine, "SuperTrend", color = trendIsUp ? color.green : color.red, linewidth=2)
plot(resistLevel, title="Resistance", color=color.red, linewidth=1, style=plot.style_circles)
plot(supportLevel, title="Support", color=color.green, linewidth=1, style=plot.style_circles)

var int tradeState = 0
validBuy  = false
validSell = false

atr = ta.atr(14)
var float buyEntry = na
var float sellEntry = na
var float stopLoss = na
var float takeProfit = na

if buySignal and tradeState != 1
    tradeState := 1
    validBuy   := true
    buyEntry   := close
    stopLoss   := close - (atr * 1.5)
    takeProfit := close + (atr * 3.0) 

if sellSignal and tradeState != -1
    tradeState := -1
    validSell  := true
    sellEntry  := close
    stopLoss   := close + (atr * 1.5)
    takeProfit := close - (atr * 3.0)

plot(tradeState == 1 ? stopLoss : na, "Long SL", color=color.red, style=plot.style_linebr, linewidth=2)
plot(tradeState == 1 ? takeProfit : na, "Long TP", color=color.blue, style=plot.style_linebr, linewidth=2)
plot(tradeState == -1 ? stopLoss : na, "Short SL", color=color.red, style=plot.style_linebr, linewidth=2)
plot(tradeState == -1 ? takeProfit : na, "Short TP", color=color.blue, style=plot.style_linebr, linewidth=2)

var label l_sl = na
var label l_tp = na
var label s_sl = na
var label s_tp = na

if not na(l_sl)
    label.delete(l_sl)
if not na(l_tp)
    label.delete(l_tp)
if not na(s_sl)
    label.delete(s_sl)
if not na(s_tp)
    label.delete(s_tp)

if tradeState == 1
    l_sl := label.new(bar_index, stopLoss, "SL", style=label.style_none, textcolor=color.red, size=size.tiny)
    l_tp := label.new(bar_index, takeProfit, "TP", style=label.style_none, textcolor=color.blue, size=size.tiny)

if tradeState == -1
    s_sl := label.new(bar_index, stopLoss, "SL", style=label.style_none, textcolor=color.red, size=size.tiny)
    s_tp := label.new(bar_index, takeProfit, "TP", style=label.style_none, textcolor=color.blue, size=size.tiny)

plotshape(validBuy,  title="Buy Signal",  style=shape.labelup,   location=location.belowbar, color=color.green, size=size.small, text="BUY", textcolor=color.white)
plotshape(validSell, title="Sell Signal", style=shape.labeldown, location=location.abovebar, color=color.red,   size=size.small, text="SELL", textcolor=color.white)

alertcondition(validBuy, title="BUY Alert", message="BUY Signal Detected")
alertcondition(validSell, title="SELL Alert", message="SELL Signal Detected")

//@version=6
indicator("PRO Intraday Strategy [Clean]", shorttitle="PRO Strategy", overlay=true, max_labels_count=500)

// -----------------------------------------------------------------------------
// Constants & Inputs
// -----------------------------------------------------------------------------
grp_st      = "SuperTrend Settings"
grp_macd    = "MACD Settings"
grp_rsi     = "RSI Settings"
grp_risk    = "Risk Management"
grp_ui      = "UI & Visuals"

stLength    = input.int(10, "ATR Length", group=grp_st)
stFactor    = input.float(3.0, "Factor", step=0.1, group=grp_st)

fastLen     = input.int(12, "Fast Length", group=grp_macd)
slowLen     = input.int(26, "Slow Length", group=grp_macd)
sigLen      = input.int(9, "Signal Length", group=grp_macd)

rsiLen      = input.int(14, "Length", group=grp_rsi)
rsiBuyLvl   = input.int(55, "Buy Threshold", group=grp_rsi)
rsiSellLvl  = input.int(45, "Sell Threshold", group=grp_rsi)

atrPeriod   = input.int(14, "ATR Period", group=grp_risk)
rrRatio     = input.float(2.0, "Risk:Reward Ratio", step=0.1, group=grp_risk)
slMultiplier= input.float(1.5, "SL ATR Multiplier", step=0.1, group=grp_risk)

colBuy      = input.color(color.new(color.teal, 0), "Buy Color", group=grp_ui)
colSell     = input.color(color.new(color.maroon, 0), "Sell Color", group=grp_ui)
showSR      = input.bool(true, "Show Support/Resistance", group=grp_ui)

// -----------------------------------------------------------------------------
// Calculations
// -----------------------------------------------------------------------------
[stLine, stDir] = ta.supertrend(stFactor, stLength)
isTrendUp       = stDir == -1
isTrendDown     = stDir == 1

[macdLine, signalLine, histLine] = ta.macd(close, fastLen, slowLen, sigLen)
isMacdBullish   = macdLine > signalLine
isMacdBearish   = macdLine < signalLine

rsiVal          = ta.rsi(close, rsiLen)

bodySize        = math.abs(close - open)
isSigCandle     = bodySize > (ta.tr(true) * 0.2) 

pivotHigh       = ta.pivothigh(high, 10, 10)
pivotLow        = ta.pivotlow(low, 10, 10)
var float rLevel = na
var float sLevel = na

if not na(pivotHigh)
    rLevel := pivotHigh
if not na(pivotLow)
    sLevel := pivotLow

// -----------------------------------------------------------------------------
// Strategy Logic
// -----------------------------------------------------------------------------
bool longCond = isTrendUp and isMacdBullish and histLine > 0 and rsiVal > rsiBuyLvl and isSigCandle
bool shortCond = isTrendDown and isMacdBearish and histLine < 0 and rsiVal < rsiSellLvl and isSigCandle

var int tradeDir = 0 // 0 = Flat, 1 = Long, -1 = Short
var float entryPrice = na
var float slPrice = na
var float tpPrice = na

atrVal = ta.atr(atrPeriod)

if tradeDir == 1
    if low <= slPrice or high >= tpPrice
        tradeDir := 0

if tradeDir == -1
    if high >= slPrice or low <= tpPrice
        tradeDir := 0

bool goLong = longCond and tradeDir == 0
bool goShort = shortCond and tradeDir == 0

if goLong
    tradeDir   := 1
    entryPrice := close
    slPrice    := close - (atrVal * slMultiplier)
    tpPrice    := close + (atrVal * slMultiplier * rrRatio)

if goShort
    tradeDir   := -1
    entryPrice := close
    slPrice    := close + (atrVal * slMultiplier)
    tpPrice    := close - (atrVal * slMultiplier * rrRatio)

// -----------------------------------------------------------------------------
// Plotting & Visuals
// -----------------------------------------------------------------------------
plot(stLine, "SuperTrend", color = isTrendUp ? colBuy : colSell, linewidth=2)
plot(showSR ? rLevel : na, "Resistance", color=color.new(colSell, 50), style=plot.style_circles)
plot(showSR ? sLevel : na, "Support",    color=color.new(colBuy, 50),  style=plot.style_circles)

plotshape(goLong,  "Long Signal",  shape.labelup,   location.belowbar, colBuy,  0, "BUY",  color.white, size=size.small)
plotshape(goShort, "Short Signal", shape.labeldown, location.abovebar, colSell, 0, "SELL", color.white, size=size.small)

plot(tradeDir == 1 ? slPrice : na, "Long SL", color.red, 1, plot.style_linebr)
plot(tradeDir == 1 ? tpPrice : na, "Long TP", color.teal, 1, plot.style_linebr)
plot(tradeDir == -1 ? slPrice : na, "Short SL", color.red, 1, plot.style_linebr)
plot(tradeDir == -1 ? tpPrice : na, "Short TP", color.teal, 1, plot.style_linebr)

bgcolor(tradeDir == 1 ? color.new(colBuy, 90) : tradeDir == -1 ? color.new(colSell, 90) : na, title="Trade Background")

alertcondition(goLong, "Buy Alert", "Long Entry Signal")
alertcondition(goShort, "Sell Alert", "Short Entry Signal")
